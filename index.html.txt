    <script>
        const tg = window.Telegram.WebApp;
        tg.ready();
        tg.expand(); // Opens the app in full-screen mode

        function showAd() {
            // Check if RichAds is loaded
            if (typeof RichAds !== 'undefined') {
                RichAds.show({
                    format: 'interstitial',
                    onSuccess: function() {
                        // This sends the reward to your Render server
                        const userId = tg.initDataUnsafe.user.id;
                        
                        // REPLACE THIS URL with your actual Render URL
                        fetch(`https://your-bot-name.onrender.com/reward?user_id=${userId}`)
                            .then(response => {
                                tg.showAlert("Success! Your balance was updated.");
                                tg.close(); // Close the app after reward
                            })
                            .catch(err => {
                                tg.showAlert("Server Error. Please try again later.");
                            });
                    },
                    onFail: function() {
                        tg.showAlert("No ads available. Try again in 5 minutes.");
                    }
                });
            } else {
                tg.showAlert("Ad blocker detected or script failed to load.");
            }
        }
